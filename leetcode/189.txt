// first approach

class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        
        int n = nums.size();

        if (n == 0 || k == 0)
        {
            return;
        }

        vector<int> temp = nums;

        for(int i = 0; i < n; i++) {
            nums[ (i + k) % n ] = temp[i];
        }
        
    }
};

// approach 2

Step1: completely reverse the whole array
Step2: now reverse the first k elements only
Step3: then reverse the left array

class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        
        int n = nums.size();
        k = k % n;

        reverse(nums.begin(), nums.end());
        reverse(nums.begin(), nums.begin() + k);
        reverse(nums.begin() + k, nums.end());
        
    }
};
